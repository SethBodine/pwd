All three generators are available via a REST API hosted on Cloudflare's edge network. Metadata (IP, browser, style) is logged to Discord — passwords are never logged.

POST /api/generate
{
  "type":      "word" | "char" | "phrase",
  "count":     1–10,

  // Word-based options
  "wordCount": 2–8,
  "separator": "-" | "." | "_" | "~" | "!" | "@" | "#" | "$" | …,
  "capitalize": true | false,
  "injectNum":  true | false,

  // Character options
  "length":  8–64,
  "lower":   true | false,
  "upper":   true | false,
  "numbers": true | false,
  "special": true | false,

  // Phrase options (uses word-based options + separator "random" picks randomly)
}
